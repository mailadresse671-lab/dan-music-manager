Verbinde deine Supabase-Datenbank, damit deine Songs auf allen Geräten verfügbar sind.


    
Schritt 1 – Tabelle in Supabase erstellen (SQL Editor)

    
CREATE TABLE IF NOT EXISTS dan_songs (
  song_id TEXT PRIMARY KEY,
  name TEXT NOT NULL,
  data JSONB NOT NULL DEFAULT '{}',
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
ALTER TABLE dan_songs ENABLE ROW LEVEL SECURITY;
CREATE POLICY "anon full access" ON dan_songs
  FOR ALL TO anon USING (true) WITH CHECK (true);
